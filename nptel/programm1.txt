#include<stdio.h>
main()
{
	int i,j,k,l,m,n,count=3;
	for(i=1;i<=7;i++)
	{
		printf("*");
	}
	printf("\n");
	for(j=1;j<=3;j++)
	{
		for(k=1;k<=count;k++)
		{
			printf("*");
		}
		count--;
		printf("\n");
	}
	printf(" ");
	count=3;
	for(l=1;l<=3;l++)
	{
		for(m=1;m<=count;m++)
		{
			printf("*");
		}
		count--;
		printf("\n");
	}
}
